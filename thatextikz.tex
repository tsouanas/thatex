\message{[ThaTeX-tikz]}

\usetikzlibrary{decorations.pathmorphing}
%\usetikzlibrary{shapes.misc}
\usetikzlibrary{positioning}

\def\TIKZPREFIX{tikz/}
\def\tikzi #1;{\input \TIKZPREFIX#1}

\tikzset{%
  dots/.style args={#1per #2}{%
    line cap=round,
    dash pattern=on 0 off #2/#1
  }
}

\def\Tikzi #1; {%
$$
\tikzpicture
\tikzi #1;
\endtikzpicture
$$%
}

\def\pointL#1{${#1}\phantom{\bullet}\phantom{#1}$}% bullet L-label
\def\pointR#1{$\phantom{#1}\phantom{\bullet}{#1}$}% bullet R-label

\def\drawmapsto#1#2;{\draw[|->] (#1) -- (#2);}

\def\drawsnake#1 #2 #3;{%
\draw [#2,decorate,decoration={snake,amplitude=.4mm,segment length=2mm,post length=1mm}]
(#1) -- (#3);
}

\def\rotator#1{%
\tikz [x=0.12cm,y=0.32cm,->,rotate=#1] \draw (0,0) arc (-150:150:1 and 1);%
}

\def\rotnodeopts#1#2#3{% with options
% #1: opts
% #2: angle
% #3: label
\node [#1] (tmp) {\phantom{#3}};
\node [#1, rotate around={#2:(tmp.center)}] at (tmp) {#3};
}
\def\rotnode#1#2{% without options
\rotnodeopts {inner sep=0pt, outer sep=0pt} {#1} {#2}
}
\def\coloredrotnode#1#2#3{% without options
\rotnodeopts {inner sep=0pt, outer sep=0pt, color=#1} {#2} {#3}
}
\def\rotated#1#2{%
\tikz \rotnode {#1} {#2};
}
\def\coloredrotated#1#2#3{%
\tikz \coloredrotnode {#1} {#2} {#3};
}
