% You must first load bussproofs.sty

\def\Assumption#1{\AxiomC{#1}}
\def\InfFromBy#1#2#3{
% write \RightLabel{#2} only for non-0 case
\ifcase#1
\AxiomC{}
\fi
% write \RightLabel{#2} in all cases
\RightLabel{#2}
\ifcase#1% 0
\UnaryInfC{#3}
\or% 1
\UnaryInfC{#3}
\or% 2
\BinaryInfC{#3}
\or% 3
\TrinaryInfC{#3}
\or% 4
\QuaternaryInfC{#3}
\or% 5
\QuinaryInfC{#3}
\fi
}
\def\Proof#1{
\def\A{\Assumption}
\def\Inf{\InfFromBy}
\def\I##1##2 ##3##4{
\InfFromBy##1{##3}{##4}
}
#1
\DisplayProof
}
\def\Proofm#1{
\def\A##1{\Assumption{$##1$}}
\def\Inf{\InfFromBy}
\def\I##1##2 ##3##4{
\InfFromBy##1{##3}{$##4$}
}
#1
\DisplayProof
}
